# models/output_task.py
from src.manager.async_task_manager import AsyncTask
from ..collectors.model_collector import ModelCollector
from ..collectors.online_time_collector import OnlineTimeCollector
from ..collectors.message_collector import MessageCollector
from ..formatters.console_formatter import ConsoleFormatter
from ..reporters.html_reporter import HTMLReporter

class StatisticOutputTask(AsyncTask):
    """统计输出任务 - 重构后的精简版本"""

    def __init__(self, record_file_path: str = "maibot_statistics.html"):
        super().__init__(
            task_name="Statistic Output Task",
            run_interval=3600
        )
        self.record_file_path = record_file_path
        self.html_reporter = HTMLReporter(record_file_path)

    async def run(self):
        """执行统计任务"""
        now = datetime.now()

        # 收集统计数据
        stats = self._collect_all_statistics(now)

        # 控制台输出
        console_output = ConsoleFormatter.format_statistics(stats)
        logger.info(f"\n{console_output}")

        # 生成 HTML 报告
        self.html_reporter.generate_report(stats, now)

    def _collect_all_statistics(self, now: datetime) -> Dict[str, Any]:
        """收集所有统计数据"""
        collect_period = self._build_collect_period(now)

        return {
            "model": ModelCollector.collect_for_period(collect_period),
            "online_time": OnlineTimeCollector.collect_for_period(collect_period, now),
            "messages": MessageCollector.collect_for_period(collect_period),
        }

#### 步骤 3: 创建路由

```python
# routes/mirror_routes.py
from fastapi import APIRouter, HTTPException
from ..models.mirror_models import MirrorAddRequest, MirrorUpdateRequest
from ..services.mirror_service import MirrorService

router = APIRouter(prefix="/mirrors", tags=["镜像源管理"])

@router.get("/")
async def get_mirrors():
    """获取所有镜像源"""
    mirrors = await MirrorService.get_all_mirrors()
    return {"mirrors": mirrors}

@router.post("/")
async def add_mirror(request: MirrorAddRequest):
    """添加镜像源"""
    try:
        result = await MirrorService.add_mirror(request)
        return {"success": True, "data": result}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.put("/{name}")
async def update_mirror(name: str, request: MirrorUpdateRequest):
    """更新镜像源"""
    success = await MirrorService.update_mirror(name, request.dict(exclude_unset=True))
    if not success:
        raise HTTPException(status_code=404, detail="镜像源不存在")
    return {"success": True}

@router.delete("/{name}")
async def delete_mirror(name: str):
    """删除镜像源"""
    success = await MirrorService.delete_mirror(name)
    if not success:
        raise HTTPException(status_code=404, detail="镜像源不存在")
    return {"success": True}
```

## 🛠️ 通用重构原则

### 4. 向后兼容
