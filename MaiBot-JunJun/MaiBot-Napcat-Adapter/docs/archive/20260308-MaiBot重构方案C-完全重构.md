# MaiBot 重构方案 C：完全重构方案

## 目标
彻底重构整个聊天架构，采用现代化的设计模式，解决所有已知问题。

## 核心设计理念

### 1. 单一 LLM 调用架构

**问题：** 当前架构需要 3 次 LLM 调用（Planner + ToolUse + Replyer）

**解决方案：** 使用 **Function Calling** 模式，一次 LLM 调用完成所有决策

```
用户消息 → 单次 LLM 调用（带工具定义） → 返回：
  - 是否需要工具
  - 需要哪些工具
  - 工具参数
  - 初步回复草稿
```

### 2. 统一的 Agent 架构

```python
# src/chat/agent/unified_agent.py

from typing import List, Dict, Any, Optional
from dataclasses import dataclass
import asyncio

@dataclass
class AgentContext:
    """Agent 上下文"""
    message: DatabaseMessages
    chat_history: List[DatabaseMessages]
    available_tools: List[Dict[str, Any]]
    bot_config: Dict[str, Any]

    # 执行结果
    tool_calls: List[ToolCall] = None
    tool_results: Dict[str, Any] = None
    reply_text: str = ""
    reasoning: str = ""


class UnifiedChatAgent:
    """统一聊天 Agent"""

    def __init__(self, chat_stream: ChatStream):
        self.chat_stream = chat_stream
        self.llm = LLMRequest(model_set=model_config.model_task_config.replyer)
        self.tool_executor = ToolExecutor(chat_id=chat_stream.stream_id)
        self.logger = get_logger("unified_agent")

    async def process(self, message: DatabaseMessages) -> Optional[str]:
        """处理消息并返回回复"""

        # 1. 构建上下文
        context = await self._build_context(message)

        # 2. 第一次 LLM 调用：决策 + 工具调用
        context = await self._llm_decision(context)

        # 3. 执行工具（如果需要）
        if context.tool_calls:
            context = await self._execute_tools(context)

            # 4. 第二次 LLM 调用：基于工具结果生成最终回复
            context = await self._llm_final_reply(context)

        # 5. 返回回复
        return context.reply_text if context.reply_text else None

    async def _build_context(self, message: DatabaseMessages) -> AgentContext:
        """构建 Agent 上下文"""
        # 获取聊天历史
        chat_history = await self.chat_stream.get_recent_messages(limit=20)

        # 获取可用工具
        available_tools = self._get_available_tools()

        # 获取机器人配置
        bot_config = {
            "name": global_config.bot.nickname,
            "personality": global_config.bot.plan_style,
            "reply_style": global_config.bot.reply_style,
        }

        return AgentContext(
            message=message,
            chat_history=chat_history,
            available_tools=available_tools,
            bot_config=bot_config
        )

    async def _llm_decision(self, context: AgentContext) -> AgentContext:
        """LLM 决策：是否需要工具、是否需要回复"""

        # 构建系统提示词
        system_prompt = f"""你是 {context.bot_config['name']}，一个智能聊天助手。

你的性格：
{context.bot_config['personality']}

你的说话风格：
{context.bot_config['reply_style']}

**重要规则：**
1. 如果需要外部信息（新闻、事件、不确定的信息），必须先调用工具获取
2. 不要编造或猜测信息
3. 如果不需要回复，返回空字符串
4. 如果需要工具，先调用工具，不要直接回复

**可用工具：**
{self._format_tools(context.available_tools)}
"""

        # 构建用户提示词
        user_prompt = self._format_chat_history(context.chat_history)
        user_prompt += f"\n\n当前消息：{context.message.sender}: {context.message.content}"

        # 调用 LLM（带工具定义）
        response, (reasoning, model_name, tool_calls) = await self.llm.generate_response_async(
            prompt=user_prompt,
            system_prompt=system_prompt,
            tools=context.available_tools,
            raise_when_empty=False
        )

        context.tool_calls = tool_calls
        context.reply_text = response
        context.reasoning = reasoning

        self.logger.info(
            f"LLM 决策完成: "
            f"tool_calls={len(tool_calls) if tool_calls else 0}, "
            f"reply={response[:50] if response else 'None'}"
        )

        return context

    async def _execute_tools(self, context: AgentContext) -> AgentContext:
        """并行执行所有工具"""
        if not context.tool_calls:
            return context

        self.logger.info(f"开始执行 {len(context.tool_calls)} 个工具")

        # 并行执行所有工具
        tasks = [
            self.tool_executor.execute_tool_call(tool_call)
            for tool_call in context.tool_calls
        ]

        results = await asyncio.gather(*tasks, return_exceptions=True)

        # 收集结果
        context.tool_results = {}
        for i, result in enumerate(results):
            if isinstance(result, Exception):
                self.logger.error(f"工具执行失败: {result}")
                continue

            tool_name = context.tool_calls[i].func_name
            context.tool_results[tool_name] = result

        self.logger.info(f"工具执行完成，成功 {len(context.tool_results)} 个")

        return context

    async def _llm_final_reply(self, context: AgentContext) -> AgentContext:
        """基于工具结果生成最终回复"""

        # 构建包含工具结果的提示词
        tool_context = self._format_tool_results(context.tool_results)

        system_prompt = f"""你是 {context.bot_config['name']}。

你刚刚调用了工具并获得了以下信息：
{tool_context}

请基于这些信息，用你的说话风格回复用户。

你的说话风格：
{context.bot_config['reply_style']}
"""

        user_prompt = f"用户问：{context.message.content}\n\n请基于工具结果回复。"

        # 第二次 LLM 调用
        response, _ = await self.llm.generate_response_async(
            prompt=user_prompt,
            system_prompt=system_prompt,
            raise_when_empty=False
        )

        context.reply_text = response

        self.logger.info(f"最终回复生成: {response[:50]}...")

        return context

    def _get_available_tools(self) -> List[Dict[str, Any]]:
        """获取可用工具定义"""
        from src.plugin_system.apis.tool_api import get_llm_available_tool_definitions
        all_tools = get_llm_available_tool_definitions()
        return [definition for name, definition in all_tools]

    def _format_tools(self, tools: List[Dict[str, Any]]) -> str:
        """格式化工具列表"""
        result = ""
        for tool in tools:
            result += f"- {tool['function']['name']}: {tool['function']['description']}\n"
        return result

    def _format_chat_history(self, history: List[DatabaseMessages]) -> str:
        """格式化聊天历史"""
        result = "**聊天历史：**\n"
        for msg in history[-10:]:  # 只取最近 10 条
            result += f"{msg.sender}: {msg.content}\n"
        return result

    def _format_tool_results(self, results: Dict[str, Any]) -> str:
        """格式化工具结果"""
        result = ""
        for tool_name, tool_result in results.items():
            content = tool_result.get("content", "")
            result += f"【{tool_name}】: {content}\n\n"
        return result
```

### 3. 简化的消息处理流程

```python
# src/chat/message_receive/bot.py

class ChatBot:
    def __init__(self):
        self.agents: Dict[str, UnifiedChatAgent] = {}

    async def message_process(self, message: MessageRecv):
        """简化的消息处理"""

        # 1. 获取或创建 Agent
        chat_id = message.chat_id
        if chat_id not in self.agents:
            chat_stream = get_chat_manager().get_stream(chat_id)
            self.agents[chat_id] = UnifiedChatAgent(chat_stream)

        agent = self.agents[chat_id]

        # 2. 转换消息格式
        db_message = self._convert_to_db_message(message)

        # 3. 调用 Agent 处理
        reply = await agent.process(db_message)

        # 4. 发送回复
        if reply:
            await self._send_reply(chat_id, reply)
```

## 架构对比

### 当前架构（7 层）
```
消息接收 → 预处理 → 心流 → 大脑循环 → Planner → Replyer → ToolUse → 发送
   ↓         ↓        ↓        ↓          ↓        ↓        ↓
 路由      命令检测   关系更新   循环管理    决策     生成     工具
```

### 新架构（3 层）
```
消息接收 → UnifiedAgent → 发送
   ↓            ↓
 路由      决策+工具+生成
```

---

## 优点

1. ✅ **极简架构**：只有 3 层，易于理解和维护
2. ✅ **减少 LLM 调用**：从 3 次减少到 1-2 次
3. ✅ **降低延迟**：流程更短，响应更快
4. ✅ **降低成本**：LLM 调用次数减少
5. ✅ **统一决策**：所有决策在一个地方完成
6. ✅ **易于测试**：Agent 是独立的，易于单元测试
7. ✅ **易于扩展**：添加新工具只需注册即可

## 缺点

1. ❌ **完全重写**：需要重写大部分代码
2. ❌ **高风险**：可能破坏现有功能
3. ❌ **需要大量测试**：确保所有功能正常
4. ❌ **迁移成本高**：需要逐步迁移现有功能
